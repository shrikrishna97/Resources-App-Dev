<template>
  <div class="admin-view">
    <AdminLogin v-if="!isLoggedIn" @login-success="handleLoginSuccess" />

    <div v-else class="admin-portal">
      <header>
        <h2>Admin Portal</h2>
        <button @click="logout">Logout</button>
      </header>

      <nav class="tabs">
        <button @click="activeTab = 'dashboard'" :class="{ active: activeTab === 'dashboard' }">Dashboard</button>
        <button @click="activeTab = 'lots'" :class="{ active: activeTab === 'lots' }">Lots</button>
        <button @click="activeTab = 'spots'" :class="{ active: activeTab === 'spots' }">Spots</button>
        <button @click="activeTab = 'users'" :class="{ active: activeTab === 'users' }">Users</button>
        <button @click="activeTab = 'reservations'" :class="{ active: activeTab === 'reservations' }">Reservations</button>
        <button @click="activeTab = 'analytics'" :class="{ active: activeTab === 'analytics' }">Analytics</button>
      </nav>

      <main>
        <AdminDashboard v-if="activeTab === 'dashboard'" />
        <AdminLots v-if="activeTab === 'lots'" />
        <AdminSpots v-if="activeTab === 'spots'" />
        <AdminUsers v-if="activeTab === 'users'" />
        <AdminReservations v-if="activeTab === 'reservations'" />
        <AdminAnalytics v-if="activeTab === 'analytics'" />
      </main>
    </div>
  </div>
</template>

<script>
import AdminLogin from '../components/AdminLogin.vue'
import AdminDashboard from '../components/AdminDashboard.vue'
import AdminLots from '../components/AdminLots.vue'
import AdminSpots from '../components/AdminSpots.vue'
import AdminUsers from '../components/AdminUsers.vue'
import AdminReservations from '../components/AdminReservations.vue'
import AdminAnalytics from '../components/AdminAnalytics.vue'

export default {
  name: 'AdminView',
  components: { AdminLogin, AdminDashboard, AdminLots, AdminSpots, AdminUsers, AdminReservations, AdminAnalytics },
  data() {
    return { isLoggedIn: false, activeTab: 'dashboard' }
  },
  mounted() { this.isLoggedIn = !!localStorage.getItem('adminToken') },
  methods: {
    handleLoginSuccess() {
      this.isLoggedIn = true
      this.activeTab = 'dashboard'
    },
    logout() {
      localStorage.removeItem('adminToken')
      this.isLoggedIn = false
    }
  }
}
</script>

<style scoped>
header { display:flex; justify-content:space-between; padding:1rem; background:#1f2937; color:#fff; }
.tabs { display:flex; gap:0.5rem; padding:0.5rem; background:#f3f4f6; flex-wrap:wrap; }
.tabs button { padding:0.5rem 1rem; border:none; background:#fff; cursor:pointer; font-size:0.9rem; }
.tabs button.active { background:#1f2937; color:#fff; }
main { padding:1rem; }
</style>
